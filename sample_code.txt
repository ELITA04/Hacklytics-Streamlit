def plot_chart():
    """
    Plotting a spider chart
    """
    fig = go.Figure(data=go.Scatterpolar(
    r=[1, 5, 2, 2, 3],
    theta=['happy','sad','anger', 'neutral','calm'],
    fill='toself'
    ))

    fig.update_layout(
    polar=dict(
        radialaxis=dict(
        visible=True
        ),
    ),
    showlegend=False
    )
    return fig
